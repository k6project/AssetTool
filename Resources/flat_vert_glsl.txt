#version 400

layout(location=0) in vec3 Position;
layout(location=1) in vec3 Normal;

layout(location=0) out vec3 FragNormal;
layout(location=1) out vec3 FragCoord;

uniform mat4 MVP;

void main()
{
    FragNormal = Normal;
    gl_Position = MVP * vec4(Position, 1.0);
    FragCoord = abs(Position);
}
